Deploy to GH Pages

Add .github\workflows\ghpages.yml to your repo:

name: Deploy GhPages
  on:
    push:
      branches:
        - main
  permissions:
    contents: write
  jobs:
    build-and-deploy:
      runs-on: ubuntu-latest
      steps:
        - uses: actions/checkout@v4
        - uses: actions/setup-java@v3
          with:
            distribution: temurin
            java-version: 11

        - name: Build
          run: ./mill site.hepek
          # or sbt hepek

        - name: Deploy
          uses: JamesIves/github-pages-deploy-action@v4
          with:
